    fn resp_rate_bpm_zero_crossing(window: &[u16]) -> Option&lt;f64&gt;:
      if window.len() &lt; 10 { return None; }
      let mean = window.iter().map(|&v| v as f64).sum::&lt;f64&gt;() / window.len() as f64;
      let centered: Vec&lt;f64&gt; = window.iter().map(|&v| v as f64 - mean).collect();
      let crossings = centered.windows(2).filter(|w| (w[0] >= 0.0) != (w[1] >= 0.0)).count();
