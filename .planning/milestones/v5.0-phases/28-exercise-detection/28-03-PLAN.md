      fn exercise_detect_sessions_bridge(request: BridgeRequest) -> BridgeResponse {
          let args: DetectExerciseSessionsArgs = match parse_args(&request) {
              Ok(a) => a,
              Err(e) => return bridge_error(&request, &e),
          };
          let store = match open_store(&args.database_path, &request) {
              Ok(s) => s,
              Err(r) => return r,
          };
          // Convert JSON types to exercise_detection types
          let hr: Vec&lt;crate::exercise_detection::HrSample&gt; = args.hr_samples.iter()
              .map(|s| crate::exercise_detection::HrSample { ts: s.ts, bpm: s.bpm })
              .collect();
          let grav: Vec&lt;crate::store::GravityRow&gt; = args.gravity_rows.iter()
              .map(|g| crate::store::GravityRow { device_id: g.device_id.clone(), ts: g.ts, x: g.x, y: g.y, z: g.z })
              .collect();
          let profile = crate::exercise_detection::ExerciseProfile {
              resting_hr: args.profile.resting_hr,
              max_hr: args.profile.max_hr,
              age: args.profile.age,
              sex: args.profile.sex.clone(),
              weight_kg: args.profile.weight_kg,
              height_cm: args.profile.height_cm,
              daily_hr_p10: args.profile.daily_hr_p10,
          };
          let sessions = crate::exercise_detection::detect_exercise_sessions(&hr, &grav, &profile);
          let mut inserted = 0usize;
          let mut warnings: Vec&lt;String&gt; = Vec::new();
          for session in &sessions {
              let row = crate::store::ExerciseSessionRow {
                  device_id: args.device_id.clone(),
                  start_ts: session.start_ts,
                  end_ts: session.end_ts,
                  duration_s: session.duration_s,
                  avg_hr: session.avg_hr,
                  peak_hr: session.peak_hr,
                  strain: session.strain,
                  calories_kcal: session.calories_kcal,
                  zone_time_pct_json: serde_json::to_string(&session.zone_time_pct).unwrap_or_default(),
                  hrmax_source: session.hrmax_source.clone(),
                  rhr_source: session.rhr_source.clone(),
                  avg_hrr_pct: session.avg_hrr_pct,
              };
              match store.insert_exercise_session(&row) {
                  Ok(true) => inserted += 1,
                  Ok(false) => warnings.push(format!("duplicate session skipped: start_ts={}", session.start_ts)),
                  Err(e) => warnings.push(format!("insert error: {e}")),
              }
          }
          bridge_ok(&request, json!({
              "sessions_detected": sessions.len(),
              "sessions_inserted": inserted,
              "warnings": warnings,
          }))
      }

      fn exercise_sessions_between_bridge(request: BridgeRequest) -> BridgeResponse {
          let args: ExerciseSessionsBetweenArgs = match parse_args(&request) {
              Ok(a) => a,
              Err(e) => return bridge_error(&request, &e),
          };
          let store = match open_store(&args.database_path, &request) {
              Ok(s) => s,
              Err(r) => return r,
          };
          match store.exercise_sessions_between(&args.device_id, args.ts_start, args.ts_end) {
              Ok(rows) => bridge_ok(&request, json!({ "sessions": rows })),
              Err(e) => bridge_error(&request, &format!("exercise_sessions_between failed: {e}")),
          }
      }
